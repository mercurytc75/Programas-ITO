import java.util.Random;

public class BubbleSortTiempo {

    public static void main(String[] args) {

        int n = 5000;
        int[] arreglo = new int[n];
        Random rand = new Random();

        // Llenar el arreglo con números aleatorios
        for (int i = 0; i < n; i++) {
            arreglo[i] = rand.nextInt(10000); // valores entre 0 y 9999
        }

        // Medir tiempo inicial
        long inicio = System.nanoTime();

        // Ordenamiento burbuja
        bubbleSort(arreglo);

        // Medir tiempo final
        long fin = System.nanoTime();

        // Calcular tiempo total en milisegundos
        double tiempoMs = (fin - inicio) / 1_000_000.0;

        System.out.println("Tiempo de ejecución (Bubble Sort): " + tiempoMs + " ms");
    }

    public static void bubbleSort(int[] arr) {
        int n = arr.length;
        boolean intercambio;

        for (int i = 0; i < n - 1; i++) {
            intercambio = false;

            for (int j = 0; j < n - i - 1; j++) {
                if (arr[j] > arr[j + 1]) {
                    // Intercambiar
                    int temp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp;
                    intercambio = true;
                }
            }

            // Optimización: si no hubo intercambio, ya está ordenado
            if (!intercambio) break;
        }
    }
}
